\begin{align}
 p = \hat{x}a \\
 e = b - p \\
 e = b - \hat{x}a \\
 a \cdot (b - \hat{x}a) = 0 \\
 a \cdot b - \hat{x}a \cdot a = 0 \\
 \hat{x} =  \frac{a \cdot b}{a\cdot a} =  \frac{a^Tb}{a^Ta}  \\
 p = \hat{x}a = \frac{a^Tb}{a^Ta} a \\
 p = a\hat{x} =  a \frac{a^Tb}{a^Ta} = Pb \\
 P = \frac{aa^T}{a^Ta}
\end{align}
